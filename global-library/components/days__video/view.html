// –®—É–∫–∞—î–º–æ –≤—Å—ñ –æ–±–≥–æ—Ä—Ç–∫–∏ (–∫–æ–Ω—Ç–µ–π–Ω–µ—Ä–∏) –ø–ª–µ—î—Ä–∞
const videoWrappers = document.querySelectorAll('.video-wrapper');

const days__video = () => {
    videoWrappers.forEach((wrapper) => {
        // –®—É–∫–∞—î–º–æ –µ–ª–µ–º–µ–Ω—Ç–∏ –¢–Ü–õ–¨–ö–ò –≤—Å–µ—Ä–µ–¥–∏–Ω—ñ –ø–æ—Ç–æ—á–Ω–æ—ó –æ–±–≥–æ—Ä—Ç–∫–∏
        const video = wrapper.querySelector('.day__video-video');
        const playBtn = wrapper.querySelector('.play-btn');

        if (!video || !playBtn) return; // –ü—Ä–æ–ø—É—Å–∫–∞—î–º–æ, —è–∫—â–æ —á–æ–≥–æ—Å—å –±—Ä–∞–∫—É—î

        // –ü–æ—á–∞—Ç–∫–æ–≤–∏–π —Å—Ç–∞–Ω
        video.controls = false;

        const togglePlay = () => {
            if (video.paused) {
                video.controls = true;
                video.play();
            } else {
                video.pause();
            }
        };

        // –ö–ª—ñ–∫ –ø–æ –∫–Ω–æ–ø—Ü—ñ
        playBtn.addEventListener('click', (e) => {
            e.stopPropagation();
            togglePlay();
        });

        // –ö–ª—ñ–∫ –ø–æ —Å–∞–º–æ–º—É –≤—ñ–¥–µ–æ (—è–∫—â–æ —Ö–æ—á–µ—à –ø–∞—É–∑—É/–ø–ª–µ–π –ø–æ –∫–ª—ñ–∫—É –Ω–∞ –∫–∞—Ä—Ç–∏–Ω–∫—É)
        video.addEventListener('click', () => {
            if (video.controls) return; // –Ø–∫—â–æ –∫–æ–Ω—Ç—Ä–æ–ª–µ—Ä–∏ –≤–∂–µ —î, –Ω–µ –∑–∞–≤–∞–∂–∞—î–º–æ
            togglePlay();
        });

        video.addEventListener('play', () => {
            playBtn.style.opacity = '0';
            playBtn.style.pointerEvents = 'none'; // –©–æ–± –∫–Ω–æ–ø–∫–∞ –Ω–µ –ø–µ—Ä–µ—Ö–æ–ø–ª—é–≤–∞–ª–∞ –∫–ª—ñ–∫–∏
            video.controls = true;
        });

        video.addEventListener('pause', () => {
            // –Ø–∫—â–æ –≤—ñ–¥–µ–æ –∑—É–ø–∏–Ω–∏–ª–∏ –Ω–∞ —Å–∞–º–æ–º—É –ø–æ—á–∞—Ç–∫—É ‚Äî –ø–æ–≤–µ—Ä—Ç–∞—î–º–æ –∫–Ω–æ–ø–∫—É
            if (video.currentTime === 0) {
                playBtn.style.opacity = '1';
                playBtn.style.pointerEvents = 'all';
                video.controls = false;
            }
        });

        video.addEventListener('ended', () => {
            video.controls = false;
            video.currentTime = 0; // –°–∫–∏–¥–∞—î–º–æ –Ω–∞ –ø–æ—á–∞—Ç–æ–∫
            playBtn.style.opacity = '1';
            playBtn.style.pointerEvents = 'all';
        });
    });
};
